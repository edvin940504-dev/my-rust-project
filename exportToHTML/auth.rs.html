use actix_web::{post, web, HttpResponse, Responder}; 
use crate::models::auth::{RegisterRequest, LoginRequest}; 

#[post("/register")]
pub async fn register_user(payload: web::Json<RegisterRequest>) -> impl Responder { 
    HttpResponse::Ok().body("Registered") 
} 

#[post("/login")]
pub async fn login_user(payload: web::Json<LoginRequest>) -> impl Responder { 
    HttpResponse::Ok().body("Logged in") 
}
